Components

A set of VueJS-based components. Use them like this: 
// Import single component to Vue
import { TwoScene } from 'https://designstem.github.io/framework/framework.js'
Vue.component('TwoScene', TwoScene)
new Vue({ el: '#app' })

// Import all framework components to Vue
import * as components from 'https://designstem.github.io/framework/framework.js'
for (const name in components) {
  Vue.component(name, components[name])
}
new Vue({ el: '#app' })


 

Utils

A set of Javascript helper functions. Use them like this:
// Import just some functions
import { random } from 'https://designstem.github.io/framework/utils.js'

// Import all under utils namespace
import * as utils from 'https://designstem.github.io/framework/utils.js'

// Make all utils available for Vue templates
new Vue({
  methods: { ...utils }
})
